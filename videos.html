firebase.initializeApp(firebaseConfig); const db = firebase.firestore(); const auth = firebase.auth(); let currentUser = null; let currentPage = 1; const perPage = 6; let allVideos = []; const videoListEl = document.getElementById("videoList"); const searchInput = document.getElementById("searchInput"); const categoryFilter = document.getElementById("categoryFilter"); enforceAccessControl((user) => { document.getElementById("userName").textContent = user.displayName || user.email; currentUser = user; loadVideos(); }); function logout() { auth.signOut().then(() => window.location.replace("/")); } function loadVideos() { db.collection("videos").get().then(snapshot => { allVideos = snapshot.docs.map(doc => ({ id: doc.id, ...doc.data() })); applyFilters(); }); } function applyFilters() { let filtered = allVideos; const searchTerm = searchInput.value.toLowerCase(); const category = categoryFilter.value; if (searchTerm) { filtered = filtered.filter(v => v.title.toLowerCase().includes(searchTerm)); } if (category) { filtered = filtered.filter(v => v.category === category); } displayVideos(filtered); } function displayVideos(videos) { const start = (currentPage - 1) * perPage; const end = start + perPage; const pageVideos = videos.slice(start, end); videoListEl.innerHTML = ""; pageVideos.forEach(video => { const card = document.createElement("div"); card.className = "video-card"; card.innerHTML = `

${video.title}

Duration: ${video.duration} | Category: ${video.category}

`; videoListEl.appendChild(card); }); document.getElementById("pageIndicator").textContent = `Page ${currentPage}`; } function submitComment(videoId) { const textArea = document.getElementById(`comment-${videoId}`); const text = textArea.value.trim(); if (!text || !currentUser) return; db.collection("comments").add({ videoId, uid: currentUser.uid, name: currentUser.displayName || currentUser.email, text, timestamp: firebase.firestore.FieldValue.serverTimestamp() }).then(() => { alert("Comment posted!"); textArea.value = ""; }); } function react(videoId, type) { if (!currentUser) return; db.collection("reactions").add({ videoId, uid: currentUser.uid, type, timestamp: firebase.firestore.FieldValue.serverTimestamp() }); } function prevPage() { if (currentPage > 1) { currentPage--; applyFilters(); } } function nextPage() { if (currentPage * perPage < allVideos.length) { currentPage++; applyFilters(); } } searchInput.addEventListener("input", applyFilters); categoryFilter.addEventListener("change", applyFilters);
